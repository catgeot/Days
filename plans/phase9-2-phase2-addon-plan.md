### Step 3: 중복 검증

```javascript
// 기존 179개와 신규 21개 비교
const existing = require('../src/pages/Home/data/travelSpots.js').TRAVEL_SPOTS;
const newDestinations = require('./phase2-addon-21.json');

const duplicates = newDestinations.filter(newDest => 
  existing.some(exist => 
    exist.name === newDest.name || 
    exist.name_en === newDest.name_en ||
    (Math.abs(exist.lat - newDest.lat) < 0.5 && Math.abs(exist.lng - newDest.lng) < 0.5)
  )
);

if (duplicates.length > 0) {
  console.error('❌ 중복 발견:', duplicates);
} else {
  console.log('✅ 중복 없음');
}
```
